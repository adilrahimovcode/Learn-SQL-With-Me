SQL Əmri Təsviri Nümunə
SELECT Verilənlər bazasından məlumatları seçmək üçün SELECT * FROM table_name;
INSERT INTO Cədvələ yeni məlumatlar əlavə etmək üçün INSERT INTO table_name (column1) VALUES (value1);
UPDATE Mövcud məlumatları yeniləmək üçün UPDATE table_name SET column1 = value1;
DELETE Məlumatları silmək üçün DELETE FROM table_name WHERE condition;
CREATE TABLE Yeni cədvəl yaratmaq üçün CREATE TABLE table_name (column1 datatype);
ALTER TABLE Mövcud cədvəli dəyişdirmək üçün ALTER TABLE table_name ADD column_name datatype;
DROP TABLE Cədvəli silmək üçün DROP TABLE table_name;
WHERE Şərt əlavə etmək üçün SELECT * FROM table_name WHERE condition;
SQL Əmri Təsviri Nümunə
ORDER BY Məlumatları sıralamaq üçün SELECT * FROM table_name ORDER BY column1;
GROUP BY Məlumatları qruplaşdırmaq üçün SELECT column1, COUNT(*) FROM table_name GROUP BY column1;
HAVING Qruplaşdırılmış nəticələrə şərt qoymaq üçün SELECT column1, COUNT(*) FROM table_name GROUP BY column1 HAVING COUNT(*) > 1;
JOIN Cədvəlləri birləşdirmək üçün SELECT * FROM table1 JOIN table2 ON table1.id = table2.id;
UNION İki və ya daha çox sorğunun nəticələrini birləşdirmək üçün SELECT column1 FROM table1 UNION SELECT column1 FROM table2;
LIMIT Nəticələrin sayını məhdudlaşdırmaq üçün SELECT * FROM table_name LIMIT 5;
IN Bir neçə dəyər arasında axtarış üçün SELECT * FROM table_name WHERE column1 IN (value1, value2);
BETWEEN İki dəyər arasında axtarış üçün SELECT * FROM table_name WHERE column1 BETWEEN value1 AND value2;
